        RemoteShell.execute({
          server: '127.0.0.1',
          port: 22,
          user: 'root',
          pass: '',
          script: 'ls',
          onOutput: function (message, type) {
            if (type === 'error') {
              console.error(message);
            } else if (type === 'info') {
              console.info(message);
            } else {
              console.log(message);
            }
          },

          },
        });
